建议：在应用程序中创建唯一一个全局变量，定义该变量为当前应用的容器

//多重嵌套
if (a) {
    if (b) {
        if (c) {
            if (d) {
                alert('所有成立');
            } else {
                alert('d不成立');
            }
        } else {
            alert('c不成立');
        }
    } else {
        alert('b不成立');
    }
} else {
    alert('a不成立');
}
//排除法
if (!a) {
    alert('a不成立');
    return;
}
if (!b) {
    alert('b不成立');
    return;
}
if (!c) {
    alert('c不成立');
    return;
}
if (!d) {
    alert('d不成立');
    return;
}
alert('所有成立');
